     func NewCmd() *cobra.Command {
         v := viper.New()
         cmd := &cobra.Command{
             Use:   "msgraph",
             Short: "Microsoft Graph API (Exchange Online) operations",
             PersistentPreRunE: func(cmd *cobra.Command, args []string) error {
                 // bind flags to viper after parse
                 if err := v.BindPFlags(cmd.Flags()); err != nil { return err }
                 BindEnvs(v)
                 return nil
             },
         }
         RegisterPersistentFlags(cmd)
         BindEnvs(v)

         cmd.AddCommand(
             newSendMailCmd(v),
             newGetEventsCmd(v),
             newGetInboxCmd(v),
             newSendInviteCmd(v),
             newGetScheduleCmd(v),
             newExportInboxCmd(v),
             newSearchAndExportCmd(v),
         )
         return cmd
     }

     Each subcommand function (e.g. newSendMailCmd) registers its
     action-specific flags (subject, body, to, cc, bcc, attachments) and its
     RunE that:
     1. Calls ConfigFromViper(v) to get *Config
     2. Calls validateConfiguration(config)
     3. Calls bootstrap.SetupSignalContext() (if not already set by root)
     4. Calls bootstrap.InitLoggers("msgraphtool", actionName, ...)
     5. Calls setupGraphClient(ctx, config, slogger)
     6. Calls the specific handler (e.g. sendEmail(...))

     Note on signal context: The root cmd's PersistentPreRunE stashes ctx/cancel
      into the cobra command's context (via cmd.SetContext(ctx)), and each RunE
     reads it via cmd.Context(). Bootstrap signal setup is called once in root's
      PersistentPreRunE.

     Step 7: Replace cmd/msgraphtool/main.go with exec shim

     The shim uses os/exec to locate and call the gomailtest binary:

     package main

     import (
         "fmt"
         "os"
         "os/exec"
         "path/filepath"
         "strings"
     )

     func main() {
         // Translate old args: msgraphtool -action=sendmail -tenantid=X ...
         //                  ‚Üí gomailtest msgraph sendmail --tenantid=X ...
         newArgs := translateArgs(os.Args[1:])

         // Find gomailtest binary: same dir as this binary first, then PATH
         exePath, _ := os.Executable()
         gomailtest := filepath.Join(filepath.Dir(exePath), "gomailtest")
         if _, err := os.Stat(gomailtest); err != nil {
             var err2 error
             gomailtest, err2 = exec.LookPath("gomailtest")
             if err2 != nil {
                 fmt.Fprintln(os.Stderr, "Error: gomailtest binary not found")
                 os.Exit(1)
             }
         }

         cmd := exec.Command(gomailtest, newArgs...)
         cmd.Stdin = os.Stdin
         cmd.Stdout = os.Stdout
         cmd.Stderr = os.Stderr
         if err := cmd.Run(); err != nil {
             os.Exit(1)
         }
     }

     func translateArgs(args []string) []string {
         // Extract -action=X or -action X to get the action name
         // Then build: ["msgraph", actionName, 
     remaining-flags-with-double-dash]
         action := "getinbox" // default
         var rest []string
         for i := 0; i < len(args); i++ {
             arg := args[i]
             if strings.HasPrefix(arg, "-action=") {
                 action = strings.TrimPrefix(arg, "-action=")
             } else if arg == "-action" && i+1 < len(args) {
                 action = args[i+1]
                 i++
             } else {
                 // Convert single-dash long flags to double-dash
                 if strings.HasPrefix(arg, "-") && !strings.HasPrefix(arg, "--")
      && len(arg) > 2 {
                     arg = "-" + arg
                 }
                 rest = append(rest, arg)
             }
         }
         return append([]string{"msgraph", action}, rest...)
     }

     Key design decisions

     1. Shim uses exec.Command, not in-process dispatch. This avoids
     restructuring cmd/gomailtest/ away from package main. The shim finds
     gomailtest in the same directory first, then PATH.
     2. Viper instance is per-protocol (created in msgraph.NewCmd()), not
     global. This keeps protocols isolated and avoids flag name collisions when
     Unit D ports other protocols.
     3. Signal context set in msgraph subcommands, not rootCmd's
     PersistentPreRunE, because root doesn't know which subcommand will run.
     Each action's RunE calls bootstrap.SetupSignalContext().
     4. validateConfiguration() stays in config.go ‚ÄĒ it takes *Config and
     returns error. The Cobra RunE calls it after ConfigFromViper().
     5. Flag names: preserve single-dash style in cobra using pflag's shorthand
     support where possible, OR use double-dash consistently in the new binary.
     The shim's translateArgs() handles the -flag ‚Üí --flag conversion for
     backward compat.
